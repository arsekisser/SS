#include "y.tab.h"
%}
%%
int|char|bool|float|void|for|do|while|if|else|return|void|main {  printf("keyword is %s\n",yytext);       return KEY;}
[+|-|*|/|=|<|>] {printf("operator is %s\n",yytext);return OP;}
[a-zA-Z][_a-zA-Z0-9]* {printf("identifier is %s\n",yytext);return ID;}
. ;
%%

YACC PART

%{
 #include <stdio.h>
 #include <stdlib.h>
 int id=0, key=0, op=0;
 %}

%token ID KEY OP 
%%
input: ID input { id++; }
 | KEY input { key++; }
 	 | OP input {op++;}
 	 | ID { id++; }
 | KEY { key++; }
 | OP { op++;}
 ;
 %%
 
extern FILE *yyin;
 
void main(int argc ,char** argv) 
{
	yyin = fopen(argv[1],"r");
	yyparse();
	printf("Keywords = %d\nIdentifiers = %d\noperators = %d\n", key,id, op);
}

void yyerror()
{
	printf("Not valid");
}









%{

        int com=0;

"//".* {com++;} 
"/*"([^*]|\*+[^*/])*\*+"/" {com++;} 
.|\n {fprintf(yyout,"%s",yytext);} 
%% 

void main(int argc, char *argv[]) 
{         
        yyin=fopen(argv[1],"r"); 
        yyout=fopen(argv[2],"w"); 
        yylex(); 
        printf("No of comment lines=%d\n",com);        
}
